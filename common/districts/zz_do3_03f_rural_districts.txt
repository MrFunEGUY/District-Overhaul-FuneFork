#03 Energy Districts

district_generator = {
	expansion_planner = yes
    base_buildtime = @do3f_rural_build_time

	min_for_deposits_on_planet = 3
	max_for_deposits_on_planet = 15
#### - Potentials -####
	show_on_uncolonized = { 
		uses_district_set = standard
		OR = {
			has_district = district_generator
			num_free_districts = { type = district_generator value > 0 }
		}
	}

	potential = { 
		uses_district_set = standard
		exists = owner
		do3f_generator_district_potential = yes
		OR = {
			has_district = district_generator
			num_free_districts = { type = district_generator value > 0 }
		}
	}
	
	allow = {
		NOR = {
			has_modifier = resort_colony
			## Megacorp_mod
			has_modifier = urban_idyll_colony
		}
	}
#### - Conversion -####
	conversion_ratio = 1
	convert_to = {
		district_generator_uncapped
		district_e_wind_turbines
		district_e_geothermal_generator
		district_e_solar_generator
		district_e_tidal_generator
		district_e_biomass_generator
		# WH40K Planets Renewed
		district_industrial_generator
		district_underdeveloped_power
	}
#### - Cost & Upkeep -####
	resources = {
		category = planet_districts
		upkeep = { energy = @do3f_base_upkeep }
		cost = {
            trigger = { pd_aw_is_megaflora_planet = no }
            minerals = @do3f_base_cost
        }
        cost = {
            trigger = { pd_aw_is_megaflora_planet = yes }
            food = @do3f_base_cost
        }
		inline_script = { script = districts/welfare_burden_upkeep_init ENERGY = @do3f_base_upkeep }
	}
####- Base -####
	planet_modifier = {
		planet_housing_add = 2
	}
# Jobs
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_hive_empire = yes }
		}
		modifier = {
			job_technician_drone_add = 3
		}
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_machine_empire = yes }
		}
		modifier = {
			job_technician_drone_add = 2
			planet_housing_add = 1
		}
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				is_gestalt = no
				is_fallen_empire_spiritualist = no
			}
		}
		modifier = {
			job_technician_add = 2
		}
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_fallen_empire_spiritualist = yes }
		}
		modifier = {
			job_fe_acolyte_generator_add = 2
		}
	}
####- Vanilla Modifiers -####
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_valid_civic = civic_agrarian_idyll }
		}
		modifier = {
			planet_housing_add = 1
		}
	}
####- ECAF -####
	inline_script = { script = districts/civic_welfare_burden_init DISTRICT_GENERATOR = 1 }
####- Stuff added by this mod -####
####- Desc -####
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		text = job_technician_drone_effect_desc
	}
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { 
				is_gestalt = no
				is_fallen_empire_spiritualist = no
			}
		}
		text = job_technician_effect_desc
	}
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_fallen_empire_spiritualist = yes }
		}
		text = job_fe_acolyte_generator_effect_desc
	}
	
	prerequisites = {
		tech_power_plant_1
	}
}

		uses_district_set = standard

			pd_aw_is_megaflora_planet = yes

		district_e_biomass_generator
		# WH40K Planets Renewed
		district_industrial_generator
		district_underdeveloped_power
	}

	resources = {
		category = planet_districts
		cost = { minerals = @do3f_base_cost }
		upkeep = { energy = @do3f_base_upkeep }
		upkeep = {
			volatile_motes = @do3f_base_upkeep
			multiplier = 0.5
		}
		produces = { energy = 5 }	
		inline_script = { script = districts/welfare_burden_upkeep_init ENERGY = @do3f_base_upkeep MOTES = 0.5 }

		planet_housing_add = 2

		potential = {
			exists = owner
			owner = { is_machine_empire = yes }
		}
		modifier = {
			job_e_fusion_drone_add = 2
			job_02_experimental_drone_add = 1
			planet_housing_add = 1
		}
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { 
				is_gestalt = no
				is_fallen_empire_spiritualist = no
			}
		}
		modifier = {
			job_e_fusion_technician_add = 2
			job_02_physicist_add = 1
		}
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_fallen_empire_spiritualist = yes }
		}
		modifier = { job_fe_acolyte_generator_add = 2 }
	}
####- Vanilla Modifiers -####
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_valid_civic = civic_agrarian_idyll }
